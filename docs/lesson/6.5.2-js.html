Nginx as Reverse Proxy and Load Balancer for Node.js

Intuition

Nginx is like a hotel concierge standing at the front door. Guests (requests) arrive and the concierge handles simple tasks directly (serving static files, redirecting). For complex requests, the concierge directs guests to the right staff member (Node.js worker). The concierge also handles security (checking IDs/SSL), manages the queue (rate limiting), and ensures VIPs (WebSocket connections) get a dedicated line.

Nginx as Reverse Proxy and Load Balancer for Node.js

Why Nginx in Front of Node.js?

Node.js can serve HTTP directly, but placing Nginx in front provides: SSL termination, static file serving, load balancing, request buffering, gzip compression, rate limiting, and security headers. Nginx handles these tasks more efficiently than Node.js.

Basic Reverse Proxy Configuration

# /etc/nginx/sites-available/myapp
server {
    listen 80;
    server_name myapp.com www.myapp.com;

    # Redirect HTTP to HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name myapp.com www.myapp.com;

    # SSL certificates
    ssl_certificate /etc/letsencrypt/live/myapp.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/myapp.com/privkey.pem;

    # SSL settings
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css application/json application/javascript
               text/xml application/xml application/xml+rss text/javascript;

    # Static files (served directly by Nginx)
    location /static/ {
        alias /var/www/myapp/public/;
        expires 1y;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    # API proxy to Node.js
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;

        # Timeouts
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # Health check endpoint (no logging)
    location /health {
        proxy_pass http://localhost:3000/health;
        access_log off;
    }

    # Block common attack paths
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }

    # Request size limit
    client_max_body_size 10M;

    # Logging
    access_log /var/log/nginx/myapp-access.log;
    error_log /var/log/nginx/myapp-error.log;
}

Load Balancing

# /etc/nginx/conf.d/upstream.conf
upstream myapp_backend {
    # Load balancing methods:
    # (default) Round Robin - distributes sequentially
    # least_conn - sends to server with fewest connections
    # ip_hash - same client always goes to same server

    least_conn;

    server 127.0.0.1:3000 weight=3;    # Gets 3x traffic
    server 127.0.0.1:3001 weight=2;
    server 127.0.0.1:3002;

    server 127.0.0.1:3003 backup;       # Only used when others are down

    # Health checks (Nginx Plus / OpenResty)
    # Or use max_fails for passive checks
    server 127.0.0.1:3000 max_fails=3 fail_timeout=30s;

    keepalive 32;    # Persistent connections to upstream
}

server {
    listen 80;

    location / {
        proxy_pass http://myapp_backend;
        proxy_http_version 1.1;
        proxy_set_header Connection "";   # Enable keepalive
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}

WebSocket Support

# WebSocket connections need special handling
location /ws {
    proxy_pass http://myapp_backend;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
    proxy_set_header Host $host;

    # Longer timeouts for WebSockets
    proxy_read_timeout 86400s;
    proxy_send_timeout 86400s;
}

Rate Limiting

# Define rate limit zones
http {
    # 10 requests per second per IP
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;

    # 1 request per second for login
    limit_req_zone $binary_remote_addr zone=login:10m rate=1r/s;

    server {
        # API rate limit (allow burst of 20)
        location /api/ {
            limit_req zone=api burst=20 nodelay;
            proxy_pass http://myapp_backend;
        }

        # Strict rate limit for auth endpoints
        location /api/auth/login {
            limit_req zone=login burst=5;
            proxy_pass http://myapp_backend;
        }
    }
}

Caching

# Define cache zone
proxy_cache_path /var/cache/nginx levels=1:2 
    keys_zone=myapp_cache:10m 
    max_size=1g 
    inactive=60m 
    use_temp_path=off;

server {
    location /api/public/ {
        proxy_pass http://myapp_backend;
        proxy_cache myapp_cache;
        proxy_cache_valid 200 10m;      # Cache 200 responses for 10 min
        proxy_cache_valid 404 1m;       # Cache 404 for 1 min
        proxy_cache_use_stale error timeout updating;
        add_header X-Cache-Status $upstream_cache_status;
    }

    # Don't cache authenticated routes
    location /api/user/ {
        proxy_pass http://myapp_backend;
        proxy_no_cache 1;               # Never cache
    }
}

Express Configuration for Nginx Proxy

// In your Express app
import express from 'express';

const app = express();

// Trust the Nginx proxy (important for req.ip, req.protocol)
app.set('trust proxy', 1);

// Now req.ip returns the real client IP (X-Forwarded-For)
// req.protocol returns 'https' (X-Forwarded-Proto)
app.get('/api/info', (req, res) => {
  res.json({
    ip: req.ip,              // Real client IP, not 127.0.0.1
    protocol: req.protocol,  // 'https', not 'http'
  });
});

Nginx Management Commands

# Test configuration (always do this before reload!)
sudo nginx -t

# Reload configuration (no downtime)
sudo nginx -s reload

# Start/stop
sudo systemctl start nginx
sudo systemctl stop nginx
sudo systemctl restart nginx

# View logs
tail -f /var/log/nginx/access.log
tail -f /var/log/nginx/error.log

# Enable a site
sudo ln -s /etc/nginx/sites-available/myapp /etc/nginx/sites-enabled/
sudo nginx -t && sudo nginx -s reload

Common Pitfalls

  1. Not setting proxy_set_header Host - Without this, your Node.js app sees 'localhost' as the host instead of the actual domain.
  2. Forgetting app.set('trust proxy', 1) in Express - Without this, req.ip returns 127.0.0.1 instead of the real client IP.
  3. Not enabling WebSocket headers - WebSocket connections fail without Upgrade and Connection headers.
  4. Not running nginx -t before reload - A syntax error in the config will take Nginx down on restart. Always test first.
  5. Serving static files through Node.js - Nginx serves static files 10-100x faster than Node.js. Offload static files to Nginx.

Exercises

  1. Production Nginx Config

    Write a complete Nginx configuration for a Node.js application with: HTTPS redirect, SSL termination, gzip compression, static file serving with cache headers, reverse proxy to Node.js, WebSocket support, rate limiting on API routes, and security headers.

    Hint

    Create separate location blocks for /static, /api, /ws, and /health. Use proxy_set_header for all necessary headers. Test with 'nginx -t'.

  2. Load Balancing Setup

    Configure Nginx to load-balance across 3 Node.js instances (ports 3000, 3001, 3002). Test with different algorithms (round-robin, least_conn, ip_hash). Use autocannon to benchmark each algorithm under load.

    Hint

    Define an upstream block with all 3 servers. Change the algorithm and benchmark with 'autocannon -c 100 -d 10 http://localhost'. Compare requests/sec.

  3. Nginx + Docker Compose

    Add Nginx as a reverse proxy to a Docker Compose setup with 2 Node.js API instances. Configure Nginx to load-balance between them, serve health checks, and handle SSL with self-signed certificates for development.

    Hint

    Use Docker Compose service names as hostnames in the upstream block. Mount the nginx.conf as a volume. Generate self-signed certs with openssl.

Common Problems & Solutions

1 502 Bad Gateway from Nginx
Cause: Nginx cannot connect to the upstream Node.js server. The app is not running, crashed, or is on a different port.
Solution: Check if Node.js is running: `pm2 status` or `curl localhost:3000`. Check Nginx error log: `tail /var/log/nginx/error.log`. Verify the proxy_pass URL matches the Node.js listening port. Check firewall rules between Nginx and Node.js.
2 WebSocket connections drop after 60 seconds
Cause: Nginx's default proxy_read_timeout is 60 seconds. Idle WebSocket connections are closed after this timeout.
Solution: Increase timeouts for WebSocket locations: `proxy_read_timeout 86400s;` (24 hours). Also set `proxy_send_timeout 86400s;`. Implement ping/pong keepalive in your WebSocket application.
3 Real client IP not available in Express (shows 127.0.0.1)
Cause: Express does not trust the proxy headers by default, so it ignores X-Forwarded-For and X-Real-IP.
Solution: Add `app.set('trust proxy', 1)` in Express. Ensure Nginx sets the headers: `proxy_set_header X-Real-IP $remote_addr;` and `proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;`.
