Project: Full CRUD API with Auth and Pagination

Introduction

In this project lesson, you'll build a production-grade REST API from scratch using NestJS. The API will be a blog platform with users, posts, comments, and tags. It will feature JWT authentication, role-based authorization, pagination, and proper error handling. This brings together everything from the previous NestJS lessons into a cohesive application.

  id          String    @id @default(uuid())
  title       String
  slug        String    @unique
  content     String
  excerpt     String?
  published   Boolean   @default(false)
  publishedAt DateTime?
  author      User      @relation(fields: [authorId], references: [id])
  authorId    String
  comments    Comment[]
  tags        Tag[]
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  @@index([slug])
  @@index([authorId])
  @@index([published, createdAt])
}

model Comment {
  id        String   @id @default(uuid())

  author    User     @relation(fields: [authorId], references: [id])
  authorId  String
  post      Post     @relation(fields: [postId], references: [id], onDelete: Cascade)
  postId    String
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@index([postId])
}

model Tag {
  id    String @id @default(uuid())

Step 3: Common DTOs and Interceptors

// common/dto/pagination-query.dto.ts
import { IsOptional, IsInt, Min, Max, IsString, IsEnum } from 'class-validator';
import { Type } from 'class-transformer';

export class PaginationQueryDto {
  @IsOptional()
  @Type(() => Number)
  @IsInt()
  @Min(1)
  page?: number = 1;

  @IsOptional()
  @Type(() => Number)
  @IsInt()
  @Min(1)
  @Max(100)
  limit?: number = 20;

  @IsOptional()
  @IsString()
  search?: string;

  @IsOptional()
  @IsString()
  sortBy?: string = 'createdAt';

  @IsOptional()
  @IsEnum(['asc', 'desc'])
  sortOrder?: 'asc' | 'desc' = 'desc';
}

// common/interceptors/transform.interceptor.ts
@Injectable()
export class TransformInterceptor<T> implements NestInterceptor<T> {
  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    return next.handle().pipe(
      map((data) => {
        // If data already has a meta property (paginated), return as-is
        if (data?.meta) return { success: true, ...data };
        return { success: true, data };
      }),
    );
  }
}

Step 4: Posts Service (Core CRUD Example)

// posts/posts.service.ts
@Injectable()
export class PostsService {
  constructor(private prisma: PrismaService) {}

  async findAll(query: PaginationQueryDto & { tag?: string; published?: boolean }) {
    const { page, limit, search, sortBy, sortOrder, tag, published } = query;

    const where: Prisma.PostWhereInput = {
      ...(published !== undefined && { published }),
      ...(search && {
        OR: [
          { title: { contains: search, mode: 'insensitive' } },
          { content: { contains: search, mode: 'insensitive' } },
        ],
      }),
      ...(tag && { tags: { some: { name: tag } } }),
    };

    const [data, total] = await Promise.all([
      this.prisma.post.findMany({
        where,
        skip: (page - 1) * limit,
        take: limit,
        orderBy: { [sortBy]: sortOrder },
        include: {
          author: { select: { id: true, name: true, avatar: true } },
          tags: { select: { id: true, name: true } },
          _count: { select: { comments: true } },
        },
      }),
      this.prisma.post.count({ where }),
    ]);

    return {
      data,
      meta: {
        total,
        page,
        limit,
        totalPages: Math.ceil(total / limit),
        hasNext: page * limit < total,
        hasPrev: page > 1,
      },
    };
  }

  async findBySlug(slug: string) {
    const post = await this.prisma.post.findUnique({
      where: { slug },
      include: {
        author: { select: { id: true, name: true, avatar: true, bio: true } },
        tags: true,
        comments: {
          orderBy: { createdAt: 'desc' },
          take: 20,
          include: {
            author: { select: { id: true, name: true, avatar: true } },
          },
        },
      },
    });

    if (!post) throw new NotFoundException('Post not found');
    return post;
  }

  async create(authorId: string, dto: CreatePostDto) {
    const slug = this.generateSlug(dto.title);

    return this.prisma.post.create({
      data: {
        title: dto.title,
        slug,
        content: dto.content,
        excerpt: dto.excerpt || dto.content.substring(0, 200),
        published: dto.published || false,
        publishedAt: dto.published ? new Date() : null,
        authorId,
        tags: dto.tagNames?.length
          ? {
              connectOrCreate: dto.tagNames.map((name) => ({
                where: { name },
                create: { name },
              })),
            }
          : undefined,
      },
      include: {
        author: { select: { id: true, name: true } },
        tags: true,
      },
    });
  }

  async update(id: string, userId: string, dto: UpdatePostDto) {
    const post = await this.prisma.post.findUnique({ where: { id } });
    if (!post) throw new NotFoundException('Post not found');
    if (post.authorId !== userId) throw new ForbiddenException('Not the author');

    return this.prisma.post.update({
      where: { id },
      data: {
        ...dto,
        slug: dto.title ? this.generateSlug(dto.title) : undefined,
        publishedAt: dto.published && !post.published ? new Date() : undefined,
        tags: dto.tagNames
          ? {
              set: [], // Remove all existing tags
              connectOrCreate: dto.tagNames.map((name) => ({
                where: { name },
                create: { name },
              })),
            }
          : undefined,
      },
      include: { tags: true },
    });
  }

  async remove(id: string, userId: string, userRole: string) {
    const post = await this.prisma.post.findUnique({ where: { id } });
    if (!post) throw new NotFoundException('Post not found');
    if (post.authorId !== userId && userRole !== 'ADMIN') {
      throw new ForbiddenException('Not authorized');
    }

    await this.prisma.post.delete({ where: { id } });
  }

  private generateSlug(title: string): string {
    return (
      title
        .toLowerCase()
        .replace(/[^a-z0-9]+/g, '-')
        .replace(/^-|-$/g, '') +
      '-' +
      Date.now().toString(36)
    );
  }
}

Step 5: Posts Controller

// posts/posts.controller.ts
@Controller('posts')
export class PostsController {
  constructor(private postsService: PostsService) {}

  @Public()
  @Get()
  findAll(@Query() query: PaginationQueryDto & { tag?: string }) {
    return this.postsService.findAll({ ...query, published: true });
  }

  @Public()
  @Get(':slug')
  findBySlug(@Param('slug') slug: string) {
    return this.postsService.findBySlug(slug);
  }

  @Post()
  create(
    @CurrentUser('id') userId: string,
    @Body() dto: CreatePostDto,
  ) {
    return this.postsService.create(userId, dto);
  }

  @Patch(':id')
  update(
    @Param('id', ParseUUIDPipe) id: string,
    @CurrentUser('id') userId: string,
    @Body() dto: UpdatePostDto,
  ) {
    return this.postsService.update(id, userId, dto);
  }

  @Delete(':id')
  @HttpCode(HttpStatus.NO_CONTENT)
  remove(
    @Param('id', ParseUUIDPipe) id: string,
    @CurrentUser() user: any,
  ) {
    return this.postsService.remove(id, user.id, user.role);
  }

  // Get current user's posts (including drafts)
  @Get('my/drafts')
  getMyDrafts(
    @CurrentUser('id') userId: string,
    @Query() query: PaginationQueryDto,
  ) {
    return this.postsService.findAll({
      ...query,
      published: false,
      authorId: userId,
    });
  }
}

Step 6: main.ts Setup

import { NestFactory } from '@nestjs/core';
import { ValidationPipe } from '@nestjs/common';
import { AppModule } from './app.module';
import { TransformInterceptor } from './common/interceptors/transform.interceptor';
import { AllExceptionsFilter } from './common/filters/all-exceptions.filter';

async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  app.setGlobalPrefix('api');

  app.useGlobalPipes(
    new ValidationPipe({
      whitelist: true,
      forbidNonWhitelisted: true,
      transform: true,
      transformOptions: { enableImplicitConversion: true },
    }),
  );

  app.useGlobalInterceptors(new TransformInterceptor());

  app.enableCors({
    origin: process.env.CORS_ORIGIN || 'http://localhost:3000',
    credentials: true,
  });

  const port = process.env.PORT || 3000;
  await app.listen(port);
  console.log(`Application running on port ${port}`);
}
bootstrap();

API Endpoints Summary

Auth:
  POST   /api/auth/register     - Register new user
  POST   /api/auth/login        - Login (returns JWT)
  POST   /api/auth/refresh      - Refresh token
  GET    /api/auth/profile      - Get current user

Posts:
  GET    /api/posts             - List published posts (public)
  GET    /api/posts/:slug       - Get post by slug (public)
  POST   /api/posts             - Create post (auth required)
  PATCH  /api/posts/:id         - Update post (author only)
  DELETE /api/posts/:id         - Delete post (author or admin)

Comments:
  GET    /api/posts/:postId/comments  - List comments
  POST   /api/posts/:postId/comments  - Add comment (auth)
  DELETE /api/comments/:id            - Delete comment (author/admin)

Tags:
  GET    /api/tags              - List all tags (public)

Users:
  GET    /api/users/:id         - Get user profile (public)
  PATCH  /api/users/:id         - Update profile (self only)
  GET    /api/admin/users       - List all users (admin only)

Common Pitfalls

  1. Circular dependency between Auth and Users modules -- Use forwardRef() or extract password hashing into a separate module.
  2. Not validating ownership -- Always check that the authenticated user owns the resource before allowing updates/deletes.
  3. Exposing passwords in responses -- Use Prisma's select to exclude password fields from all queries.
  4. Missing pagination on list endpoints -- Unbounded queries will crash with large datasets.

Exercises

  1. Build the Complete Blog API

    Follow this lesson step by step to build the full blog API. Start with the Prisma schema, then build each module: Auth, Users, Posts, Comments, Tags. Add global validation, error handling, and response transformation.

    Hint

    Work module by module. Start with Prisma + Auth (you need auth for everything else). Then Posts, Comments, Tags. Test each module with curl or Postman before moving on.

  2. Add Post Search with Full-Text

    Enhance the posts list endpoint with advanced search: full-text search across title and content, filter by multiple tags (AND logic), filter by date range, and sort by relevance when searching.

    Hint

    Use Prisma's search mode for full-text search on PostgreSQL. For multi-tag filtering, use tags: { every: { name: { in: tagNames } } }. Add startDate/endDate query params.

  3. Implement Soft Delete

    Add soft delete support to posts and comments. Records should have a deletedAt field. All queries should automatically exclude soft-deleted records. Add an admin endpoint to permanently delete or restore soft-deleted records.

    Hint

    Add deletedAt DateTime? to the Prisma schema. Use Prisma middleware to automatically filter where deletedAt: null. Create admin endpoints for hard delete and restore.

Common Problems & Solutions

1 Slug collisions when two posts have the same title
Cause: The slug generation doesn't account for uniqueness beyond the title text.
Solution: Append a unique suffix to the slug (timestamp in base36, random string, or UUID segment). Check for existing slugs and retry with a different suffix if needed.
2 N+1 queries on the posts list
Cause: Fetching posts and then separately fetching author, tags, and comment counts for each post.
Solution: Use Prisma's `include` or `select` with `_count` to load all related data in a single query. The findAll method should include author, tags, and _count in one query.
3 Comments can be added to unpublished posts
Cause: The comment creation endpoint doesn't check if the parent post is published.
Solution: Before creating a comment, fetch the post and verify it's published. Or add a Prisma middleware that checks this condition.
