Mongoose: Schemas, Models, Validation, and Middleware

Intuition

Mongoose is like a security guard and personal assistant for your MongoDB data. Without it, anyone can put anything in any format into the database. Mongoose checks every document against the schema rules (the guard), transforms data automatically (lowercase emails, trim whitespace), and runs custom logic on save/delete (the assistant). It trades some of MongoDB's flexibility for safety and convenience.

Mongoose: Schemas, Models, Validation, and Middleware

Introduction

Mongoose is an ODM (Object Document Mapper) for MongoDB and Node.js. While you can use the native MongoDB driver directly, Mongoose adds schemas, validation, middleware (hooks), type casting, and an elegant query API. It brings structure to MongoDB's flexibility — enforcing rules at the application level that MongoDB doesn't enforce at the database level.

Setup

npm install mongoose

import mongoose from 'mongoose';

await mongoose.connect(process.env.MONGODB_URI || 'mongodb://localhost:27017/myapp', {
  maxPoolSize: 10,
  serverSelectionTimeoutMS: 5000,
});

mongoose.connection.on('connected', () => console.log('Mongoose connected'));
mongoose.connection.on('error', (err) => console.error('Mongoose error:', err));

Defining Schemas

import { Schema, model, Document, Types } from 'mongoose';

// Interface for TypeScript
interface IUser extends Document {
  name: string;
  email: string;
  passwordHash: string;
  role: 'customer' | 'admin' | 'vendor';
  age?: number;
  tags: string[];
  address?: {
    street: string;
    city: string;
    state: string;
    zip: string;
  };
  preferences: Record<string, any>;
  createdAt: Date;
  updatedAt: Date;
}

const userSchema = new Schema<IUser>({
  name: {
    type: String,
    required: [true, 'Name is required'],
    trim: true,
    minlength: [2, 'Name must be at least 2 characters'],
    maxlength: [100, 'Name must be under 100 characters'],
  },
  email: {
    type: String,
    required: [true, 'Email is required'],
    unique: true,
    lowercase: true,
    trim: true,
    match: [/^\S+@\S+\.\S+$/, 'Invalid email format'],
  },
  passwordHash: {
    type: String,
    required: true,
    select: false, // excluded from queries by default
  },
  role: {
    type: String,
    enum: ['customer', 'admin', 'vendor'],
    default: 'customer',
  },
  age: {
    type: Number,
    min: [13, 'Must be at least 13'],
    max: [150, 'Invalid age'],
  },
  tags: [{ type: String, trim: true }],
  address: {
    street: String,
    city: String,
    state: String,
    zip: String,
  },
  preferences: {
    type: Schema.Types.Mixed,
    default: {},
  },
}, {
  timestamps: true,        // auto createdAt, updatedAt
  toJSON: { virtuals: true },
  toObject: { virtuals: true },
});

// Indexes
userSchema.index({ email: 1 }, { unique: true });
userSchema.index({ role: 1, createdAt: -1 });
userSchema.index({ name: 'text', email: 'text' }); // text search

Virtuals

// Computed properties (not stored in DB)
userSchema.virtual('isAdmin').get(function() {
  return this.role === 'admin';
});

userSchema.virtual('displayName').get(function() {
  return `${this.name} (${this.email})`;
});

// Virtual populate (like a JOIN)
userSchema.virtual('orders', {
  ref: 'Order',
  localField: '_id',
  foreignField: 'userId',
});

Instance Methods and Statics

// Instance method (on a document)
userSchema.methods.comparePassword = async function(password: string) {
  const user = await User.findById(this._id).select('+passwordHash');
  return bcrypt.compare(password, user!.passwordHash);
};

userSchema.methods.toSafeObject = function() {
  const obj = this.toObject();
  delete obj.passwordHash;
  return obj;
};

// Static method (on the model)
userSchema.statics.findByEmail = function(email: string) {
  return this.findOne({ email: email.toLowerCase() });
};

userSchema.statics.findActive = function() {
  return this.find({ deletedAt: null });
};

Middleware (Hooks)

// Pre-save: runs before save()
userSchema.pre('save', async function(next) {
  // Hash password if modified
  if (this.isModified('passwordHash')) {
    this.passwordHash = await bcrypt.hash(this.passwordHash, 10);
  }
  next();
});

// Pre-find: auto-exclude soft-deleted
userSchema.pre(/^find/, function(next) {
  // 'this' is the query
  this.where({ deletedAt: null });
  next();
});

// Post-save: runs after save()
userSchema.post('save', function(doc) {
  console.log('User saved:', doc.email);
});

// Pre-remove
userSchema.pre('deleteOne', { document: true, query: false }, async function(next) {
  // Clean up related data
  await mongoose.model('Order').deleteMany({ userId: this._id });
  next();
});

Creating the Model

const User = model<IUser>('User', userSchema);
export default User;

CRUD with Mongoose

// Create
const user = await User.create({
  name: 'Alice',
  email: 'alice@example.com',

});

// Read
const user = await User.findById(id);
const user = await User.findOne({ email: 'alice@example.com' });
const users = await User.find({ role: 'customer' })
  .sort({ createdAt: -1 })
  .limit(20)
  .skip(0)
  .select('name email role');

// With populate (like JOIN)
const userWithOrders = await User.findById(id)
  .populate({
    path: 'orders',
    options: { sort: { createdAt: -1 }, limit: 10 },
    select: 'total status createdAt',
  });

// Update
await User.findByIdAndUpdate(id, { name: 'New Name' }, {
  new: true,        // return updated document
  runValidators: true, // run schema validators
});

// Delete
await User.findByIdAndDelete(id);

Validation

const productSchema = new Schema({
  name: {
    type: String,
    required: true,
    validate: {
      validator: (v: string) => v.length >= 2,
      message: 'Name must be at least 2 characters',
    },
  },
  price: {
    type: Number,
    required: true,
    validate: {
      validator: (v: number) => v >= 0,
      message: 'Price must be non-negative',
    },
  },
  sku: {
    type: String,
    required: true,
    validate: {
      validator: async function(v: string) {
        const existing = await mongoose.model('Product').findOne({ sku: v });
        return !existing || existing._id.equals(this._id);
      },
      message: 'SKU must be unique',
    },
  },
});

// Handling validation errors
try {
  await Product.create({ name: '', price: -5 });
} catch (err) {
  if (err instanceof mongoose.Error.ValidationError) {
    const errors = Object.values(err.errors).map(e => ({
      field: e.path,
      message: e.message,
    }));
    console.log(errors);
    // [{ field: 'name', message: 'Name must be at least 2 characters' },
    //  { field: 'price', message: 'Price must be non-negative' }]
  }
}

Relations with Populate

const orderSchema = new Schema({
  userId: { type: Schema.Types.ObjectId, ref: 'User', required: true },
  items: [{
    productId: { type: Schema.Types.ObjectId, ref: 'Product', required: true },
    quantity: { type: Number, required: true, min: 1 },
    price: { type: Number, required: true },
  }],
  total: { type: Number, required: true },
  status: {
    type: String,
    enum: ['pending', 'processing', 'shipped', 'delivered', 'cancelled'],
    default: 'pending',
  },
}, { timestamps: true });

// Deep populate
const order = await Order.findById(orderId)
  .populate('userId', 'name email')  // only name and email from user
  .populate('items.productId', 'name price images');

Graceful Shutdown

async function shutdown() {
  await mongoose.connection.close();
  console.log('Mongoose disconnected');
  process.exit(0);
}

process.on('SIGTERM', shutdown);
process.on('SIGINT', shutdown);

Common Problems & Solutions

1 Populate returns null even though the referenced document exists
Cause: The ref value in the schema doesn't match the model name, or the stored ID is a string instead of an ObjectId.
Solution: Verify that `ref: 'ModelName'` matches exactly what you passed to `model('ModelName', schema)`. Check that IDs are stored as ObjectId, not strings.
2 Validation doesn't run on findByIdAndUpdate
Cause: By default, Mongoose doesn't run validators on update operations. Only save() triggers validators.
Solution: Pass `{ runValidators: true }` as an option: `findByIdAndUpdate(id, update, { runValidators: true })`.
3 Pre-save middleware runs but this.isModified() is always true
Cause: Using create() or save() on a new document — all fields are considered 'modified' on the first save.
Solution: Check `this.isNew` to distinguish between create and update. For password hashing, combine: `if (this.isModified('password') && !this.isNew) { ... }` — or simply always hash if modified.
