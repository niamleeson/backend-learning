Upload Configuration

// common/config/multer.config.ts
import { diskStorage } from 'multer';
import { extname } from 'path';
import { BadRequestException } from '@nestjs/common';

export const multerConfig = {
  storage: diskStorage({
    destination: './uploads',
    filename: (req, file, callback) => {
      const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1e9);
      const ext = extname(file.originalname);
      callback(null, `${uniqueSuffix}${ext}`);
    },
  }),
  fileFilter: (req: any, file: any, callback: any) => {
    if (!file.mimetype.match(/\/(jpg|jpeg|png|gif|webp)$/)) {
      return callback(
        new BadRequestException('Only image files are allowed'),
        false,
      );
    }
    callback(null, true);
  },
  limits: {
    fileSize: 5 * 1024 * 1024, // 5MB
  },
};

Upload Endpoint

// users/users.controller.ts
import { FileInterceptor } from '@nestjs/platform-express';

@Controller('users')
export class UsersController {
  @Patch('avatar')
  @UseInterceptors(FileInterceptor('avatar', multerConfig))
  async uploadAvatar(
    @CurrentUser('id') userId: string,
    @UploadedFile() file: Express.Multer.File,
  ) {
    if (!file) throw new BadRequestException('No file uploaded');

    const avatarUrl = `/uploads/${file.filename}`;
    return this.usersService.updateAvatar(userId, avatarUrl);
  }
}

S3 Uploads (Production)

// common/services/s3-upload.service.ts
import { Injectable } from '@nestjs/common';
import { S3Client, PutObjectCommand } from '@aws-sdk/client-s3';
import { ConfigService } from '@nestjs/config';
import { v4 as uuidv4 } from 'uuid';

@Injectable()
export class S3UploadService {
  private s3: S3Client;
  private bucket: string;

  constructor(private config: ConfigService) {
    this.s3 = new S3Client({
      region: config.get('AWS_REGION'),
      credentials: {
        accessKeyId: config.get('AWS_ACCESS_KEY_ID'),
        secretAccessKey: config.get('AWS_SECRET_ACCESS_KEY'),
      },
    });
    this.bucket = config.get('S3_BUCKET');
  }

  async upload(file: Express.Multer.File, folder: string): Promise<string> {
    const key = `${folder}/${uuidv4()}-${file.originalname}`;

    await this.s3.send(
      new PutObjectCommand({
        Bucket: this.bucket,
        Key: key,
        Body: file.buffer,
        ContentType: file.mimetype,
      }),
    );

    return `https://${this.bucket}.s3.amazonaws.com/${key}`;
  }
}

Advanced Filtering

// posts/dto/filter-posts.dto.ts
export class FilterPostsDto extends PaginationQueryDto {
  @IsOptional()
  @IsString()
  search?: string;

  @IsOptional()
  @IsString()
  authorId?: string;

  @IsOptional()
  @Transform(({ value }) => value === 'true')
  @IsBoolean()
  published?: boolean;

  @IsOptional()
  @Transform(({ value }) =>
    typeof value === 'string' ? value.split(',') : value,
  )
  @IsArray()
  @IsString({ each: true })
  tags?: string[];

  @IsOptional()
  @Type(() => Date)
  @IsDate()
  startDate?: Date;

  @IsOptional()
  @Type(() => Date)
  @IsDate()
  endDate?: Date;

  @IsOptional()
  @IsEnum(['title', 'createdAt', 'updatedAt'])
  sortBy?: string = 'createdAt';
}

// posts/posts.service.ts
async findAll(query: FilterPostsDto) {
  const where: Prisma.PostWhereInput = {
    AND: [
      query.published !== undefined ? { published: query.published } : {},
      query.authorId ? { authorId: query.authorId } : {},
      query.search
        ? {
            OR: [
              { title: { contains: query.search, mode: 'insensitive' } },
              { content: { contains: query.search, mode: 'insensitive' } },
              { author: { name: { contains: query.search, mode: 'insensitive' } } },
            ],
          }
        : {},
      query.tags?.length
        ? { tags: { some: { name: { in: query.tags } } } }
        : {},
      query.startDate || query.endDate
        ? {
            createdAt: {
              ...(query.startDate && { gte: query.startDate }),
              ...(query.endDate && { lte: query.endDate }),
            },
          }
        : {},
    ],
  };

  const [data, total] = await Promise.all([
    this.prisma.post.findMany({
      where,
      skip: (query.page - 1) * query.limit,
      take: query.limit,
      orderBy: { [query.sortBy]: query.sortOrder },
      include: {
        author: { select: { id: true, name: true, avatar: true } },
        tags: { select: { id: true, name: true } },
        _count: { select: { comments: true } },
      },
    }),
    this.prisma.post.count({ where }),
  ]);

  return {
    data,
    meta: {
      total,
      page: query.page,
      limit: query.limit,
      totalPages: Math.ceil(total / query.limit),
    },
  };
}

Rate Limiting

npm i @nestjs/throttler

// app.module.ts
import { ThrottlerModule, ThrottlerGuard } from '@nestjs/throttler';

@Module({
  imports: [
    ThrottlerModule.forRoot([
      { name: 'short', ttl: 1000, limit: 3 },   // 3 requests per second

      { name: 'long', ttl: 60000, limit: 100 },  // 100 per minute
    ]),

  providers: [
    { provide: APP_GUARD, useClass: ThrottlerGuard },
  ],
})

Response Caching

npm i @nestjs/cache-manager cache-manager

// app.module.ts
import { CacheModule } from '@nestjs/cache-manager';

@Module({
  imports: [
    CacheModule.register({
      ttl: 60, // seconds
      max: 100, // max items in cache
      isGlobal: true,
    }),
  ],
})
export class AppModule {}

// posts/posts.controller.ts
import { CacheInterceptor, CacheTTL, CacheKey } from '@nestjs/cache-manager';

@Controller('posts')
export class PostsController {
  @UseInterceptors(CacheInterceptor)
  @CacheTTL(30)
  @Get()
  findAll(@Query() query: FilterPostsDto) {
    return this.postsService.findAll(query);
  }

  // Manual cache invalidation
  constructor(@Inject(CACHE_MANAGER) private cacheManager: Cache) {}

  @Post()
  async create(@CurrentUser('id') userId: string, @Body() dto: CreatePostDto) {
    const post = await this.postsService.create(userId, dto);
    await this.cacheManager.reset(); // Clear all cache
    return post;
  }
}

Pagination Metadata

export class PaginatedResponse<T> {
  data: T[];
  meta: {
    total: number;
    page: number;
    limit: number;
    totalPages: number;
    hasNextPage: boolean;
    hasPreviousPage: boolean;
  };
  links: {
    first: string;
    last: string;
    next: string | null;
    previous: string | null;
  };
}

// Helper function
function buildPaginationLinks(
  baseUrl: string,
  page: number,
  limit: number,
  totalPages: number,
): PaginatedResponse<any>['links'] {
  return {
    first: `${baseUrl}?page=1&limit=${limit}`,
    last: `${baseUrl}?page=${totalPages}&limit=${limit}`,
    next: page < totalPages ? `${baseUrl}?page=${page + 1}&limit=${limit}` : null,
    previous: page > 1 ? `${baseUrl}?page=${page - 1}&limit=${limit}` : null,
  };
}

Common Pitfalls

  1. File uploads without validation -- Always validate file type, size, and content. Don't trust the file extension alone.
  2. Storing uploads on the app server -- Use S3 or similar for production. Local storage doesn't work with multiple server instances.
  3. Caching authenticated responses -- The cache interceptor might serve User A's data to User B. Use cache keys that include user identity for personalized data.
  4. Rate limiting WebSocket connections -- @nestjs/throttler only works with HTTP by default. WebSockets need custom rate limiting.

Exercises

  1. Add File Uploads to the Blog API

    Implement avatar upload for users and image upload for posts. Validate file types (images only) and sizes (max 5MB). Store files locally in development and document how to switch to S3 for production.

    Hint

    Use FileInterceptor with multer config for single files. Create a reusable multer config with fileFilter for image validation. Add the upload path to .gitignore.

  2. Implement Advanced Filtering

    Add comprehensive filtering to the posts endpoint: search across title/content/author, filter by tags (comma-separated), date range, published status. Support multiple sort fields. Create a reusable filter builder.

    Hint

    Use @Transform() in DTOs to convert comma-separated strings to arrays. Build the Prisma where clause dynamically based on which filters are provided.

  3. Add Rate Limiting and Caching

    Configure @nestjs/throttler with different limits for different endpoints (stricter for auth, relaxed for reads). Add response caching for public endpoints with cache invalidation on writes.

    Hint

    Use the multi-throttler config with named throttlers. Apply @Throttle() per endpoint for custom limits. Use CACHE_MANAGER to manually invalidate cache after creates/updates.

Common Problems & Solutions

1 Uploaded file is missing after server restart
Cause: Files are stored in the container's filesystem. Container restarts wipe the filesystem.
Solution: Use external storage (S3, GCS) for production. For local development, mount a volume. Never store uploads in the application directory.
2 Cache returns stale data after updates
Cause: The cache TTL hasn't expired, so the old data is served even though the database was updated.
Solution: Manually invalidate cache after write operations using `cacheManager.del(key)` or `cacheManager.reset()`. Use cache keys that include relevant filter parameters.
3 Rate limiter blocks legitimate users
Cause: Rate limits are too restrictive, or all users behind a load balancer share the same IP.
Solution: Tune rate limits based on actual usage patterns. Use X-Forwarded-For header to get the real client IP behind proxies. Consider user-based rate limiting instead of IP-based for authenticated endpoints.
