Building a Real-Time Chat Application

Intuition

Building a chat app with WebSockets is like setting up a conference call system. Each room is a separate call line. When someone speaks (sends a message), everyone on that line hears it. The gateway is the phone switch that routes calls, the service is the recording system that saves the conversation, and Socket.IO rooms are the phone lines themselves.

Building a Real-Time Chat Application

Introduction

This project lesson guides you through building a real-time chat application using NestJS WebSockets (Socket.IO), authentication, and database persistence. You'll implement user authentication for WebSocket connections, chat rooms, message broadcasting, message history, and typing indicators.

Data Models

model User {
  id        String    @id @default(uuid())
  name      String
  email     String    @unique
  password  String
  messages  Message[]
  rooms     RoomMember[]
  createdAt DateTime  @default(now())
}

model Room {
  id        String       @id @default(uuid())
  name      String
  isPrivate Boolean      @default(false)
  members   RoomMember[]
  messages  Message[]
  createdAt DateTime     @default(now())
}

model RoomMember {
  id       String   @id @default(uuid())
  user     User     @relation(fields: [userId], references: [id])
  userId   String
  room     Room     @relation(fields: [roomId], references: [id])
  roomId   String
  role     String   @default("member")
  joinedAt DateTime @default(now())

  @@unique([userId, roomId])
}

model Message {
  id        String   @id @default(uuid())
  content   String
  author    User     @relation(fields: [authorId], references: [id])
  authorId  String
  room      Room     @relation(fields: [roomId], references: [id])
  roomId    String
  createdAt DateTime @default(now())

  @@index([roomId, createdAt])
}

Chat Gateway

// chat/chat.gateway.ts
import {
  WebSocketGateway, WebSocketServer, SubscribeMessage,
  MessageBody, ConnectedSocket, OnGatewayConnection,
  OnGatewayDisconnect,
} from '@nestjs/websockets';
import { Server, Socket } from 'socket.io';
import { UseGuards, Logger } from '@nestjs/common';
import { WsAuthGuard } from '../auth/guards/ws-auth.guard';
import { ChatService } from './chat.service';

interface AuthenticatedSocket extends Socket {
  data: { user: { id: string; name: string; email: string } };
}

@WebSocketGateway({ cors: { origin: '*' } })
@UseGuards(WsAuthGuard)
export class ChatGateway implements OnGatewayConnection, OnGatewayDisconnect {
  @WebSocketServer()
  server: Server;

  private logger = new Logger('ChatGateway');
  private userSockets = new Map<string, Set<string>>(); // userId -> socketIds

  constructor(private chatService: ChatService) {}

  async handleConnection(client: AuthenticatedSocket) {
    try {
      const user = client.data.user;
      this.logger.log(`User ${user.name} connected (${client.id})`);

      // Track user's sockets (supports multiple tabs)
      if (!this.userSockets.has(user.id)) {
        this.userSockets.set(user.id, new Set());
      }
      this.userSockets.get(user.id).add(client.id);

      // Join user's personal room for direct messages
      client.join(`user:${user.id}`);

      // Join all rooms the user is a member of
      const rooms = await this.chatService.getUserRooms(user.id);
      for (const room of rooms) {
        client.join(`room:${room.id}`);
      }

      // Notify rooms about user coming online
      this.server.emit('user-online', { userId: user.id, name: user.name });
    } catch (error) {
      this.logger.error(`Connection error: ${error.message}`);
      client.disconnect();
    }
  }

  async handleDisconnect(client: AuthenticatedSocket) {
    const user = client.data?.user;
    if (!user) return;

    this.userSockets.get(user.id)?.delete(client.id);

    // Only emit offline if user has no more connected sockets
    if (this.userSockets.get(user.id)?.size === 0) {
      this.userSockets.delete(user.id);
      this.server.emit('user-offline', { userId: user.id });
    }

    this.logger.log(`User ${user.name} disconnected (${client.id})`);
  }

  @SubscribeMessage('send-message')
  async handleMessage(
    @MessageBody() data: { roomId: string; content: string },
    @ConnectedSocket() client: AuthenticatedSocket,
  ) {
    const user = client.data.user;

    // Persist message
    const message = await this.chatService.createMessage({
      content: data.content,
      roomId: data.roomId,
      authorId: user.id,
    });

    // Broadcast to room
    this.server.to(`room:${data.roomId}`).emit('new-message', {
      id: message.id,
      content: message.content,
      roomId: data.roomId,
      author: { id: user.id, name: user.name },
      createdAt: message.createdAt,
    });

    return { status: 'ok', messageId: message.id };
  }

  @SubscribeMessage('join-room')
  async handleJoinRoom(
    @MessageBody() data: { roomId: string },
    @ConnectedSocket() client: AuthenticatedSocket,
  ) {
    const user = client.data.user;
    await this.chatService.joinRoom(user.id, data.roomId);
    client.join(`room:${data.roomId}`);

    // Notify room members
    this.server.to(`room:${data.roomId}`).emit('user-joined', {
      userId: user.id,
      name: user.name,
      roomId: data.roomId,
    });

    // Send recent message history to the new member
    const messages = await this.chatService.getMessages(data.roomId, 50);
    client.emit('message-history', { roomId: data.roomId, messages });

    return { status: 'ok' };
  }

  @SubscribeMessage('leave-room')
  async handleLeaveRoom(
    @MessageBody() data: { roomId: string },
    @ConnectedSocket() client: AuthenticatedSocket,
  ) {
    const user = client.data.user;
    await this.chatService.leaveRoom(user.id, data.roomId);
    client.leave(`room:${data.roomId}`);

    this.server.to(`room:${data.roomId}`).emit('user-left', {
      userId: user.id,
      name: user.name,
      roomId: data.roomId,
    });

    return { status: 'ok' };
  }

  @SubscribeMessage('typing')
  handleTyping(
    @MessageBody() data: { roomId: string },
    @ConnectedSocket() client: AuthenticatedSocket,
  ) {
    const user = client.data.user;
    client.to(`room:${data.roomId}`).emit('user-typing', {
      userId: user.id,
      name: user.name,
      roomId: data.roomId,
    });
  }

  @SubscribeMessage('stop-typing')
  handleStopTyping(
    @MessageBody() data: { roomId: string },
    @ConnectedSocket() client: AuthenticatedSocket,
  ) {
    const user = client.data.user;
    client.to(`room:${data.roomId}`).emit('user-stop-typing', {
      userId: user.id,
      roomId: data.roomId,
    });
  }

  // Helper: check if user is online
  isUserOnline(userId: string): boolean {
    return this.userSockets.has(userId) && this.userSockets.get(userId).size > 0;
  }

  getOnlineUsers(): string[] {
    return Array.from(this.userSockets.keys());
  }
}

Chat Service

// chat/chat.service.ts
@Injectable()
export class ChatService {
  constructor(private prisma: PrismaService) {}

  async createMessage(data: { content: string; roomId: string; authorId: string }) {
    return this.prisma.message.create({
      data,
      include: { author: { select: { id: true, name: true } } },
    });
  }

  async getMessages(roomId: string, limit = 50, cursor?: string) {
    return this.prisma.message.findMany({
      where: { roomId },
      take: limit,
      ...(cursor && { cursor: { id: cursor }, skip: 1 }),
      orderBy: { createdAt: 'desc' },
      include: { author: { select: { id: true, name: true } } },
    });
  }

  async createRoom(name: string, creatorId: string, isPrivate = false) {
    return this.prisma.room.create({
      data: {
        name,
        isPrivate,
        members: {
          create: { userId: creatorId, role: 'admin' },
        },
      },
    });
  }

  async joinRoom(userId: string, roomId: string) {
    return this.prisma.roomMember.upsert({
      where: { userId_roomId: { userId, roomId } },
      create: { userId, roomId },
      update: {},
    });
  }

  async leaveRoom(userId: string, roomId: string) {
    await this.prisma.roomMember.delete({
      where: { userId_roomId: { userId, roomId } },
    });
  }

  async getUserRooms(userId: string) {
    const memberships = await this.prisma.roomMember.findMany({
      where: { userId },
      include: { room: true },
    });
    return memberships.map((m) => m.room);
  }

  async getRoomMembers(roomId: string) {
    const memberships = await this.prisma.roomMember.findMany({
      where: { roomId },
      include: { user: { select: { id: true, name: true, email: true } } },
    });
    return memberships.map((m) => ({ ...m.user, role: m.role }));
  }
}

Event Summary

Client -> Server Events:
  send-message   { roomId, content }
  join-room      { roomId }
  leave-room     { roomId }
  typing         { roomId }
  stop-typing    { roomId }

Server -> Client Events:
  new-message    { id, content, roomId, author, createdAt }
  user-joined    { userId, name, roomId }
  user-left      { userId, name, roomId }
  user-typing    { userId, name, roomId }
  user-stop-typing { userId, roomId }
  user-online    { userId, name }
  user-offline   { userId }
  message-history { roomId, messages[] }
  error          { message }

Common Pitfalls

  1. Not handling multiple tabs -- A user can have multiple socket connections. Track sockets per user, and only emit 'offline' when all sockets disconnect.
  2. Message ordering -- Network delays can cause messages to arrive out of order. Always include timestamps and sort client-side.
  3. Memory leaks in userSockets map -- Always clean up the map in handleDisconnect.
  4. Missing room authorization -- Verify the user is a member of the room before allowing messages or access to history.

  1. Build the Complete Chat Application

    Hint

    Start with the Prisma schema and run migrations. Then build the ChatService, ChatGateway, and ChatController. Test with a Socket.IO client or a simple HTML page.

  2. Add Direct Messaging

    Extend the chat system to support direct messages between two users. When a user sends a DM, automatically create a private room for the two users if one doesn't exist. Show DM conversations separately from group rooms.

    Hint

    Create rooms with isPrivate: true and exactly two members. Generate a deterministic room name from both user IDs (sort alphabetically) to find existing DM rooms.

  3. Implement Message Read Receipts

    Add read receipts: when a user reads messages in a room, notify the message authors that their messages were read. Track the last-read message per user per room. Show unread count for each room.

    Hint

    Add a ReadReceipt model or a lastReadMessageId to RoomMember. When a user emits 'mark-read', update the database and notify the room. Calculate unread count from messages after lastReadMessageId.

Common Problems & Solutions

1 Messages are duplicated when user has multiple tabs open
Cause: Each tab creates a separate socket connection that joins the same room, so the user receives the message once per tab from the server, plus displays it locally.
Solution: On the client side, deduplicate messages by ID. On the server side, this is expected behavior -- each socket in the room receives the event.
2 Message history is missing for new room members
Cause: Messages are only broadcast to currently connected clients. A user who joins later doesn't see past messages.
Solution: When a user joins a room (join-room event), fetch recent messages from the database and send them as 'message-history'. Use cursor-based pagination for loading more.
3 User appears online after closing all tabs
Cause: The handleDisconnect didn't properly clean up the userSockets map, or the disconnect event wasn't fired (network issue).
Solution: Use Socket.IO's pingTimeout/pingInterval to detect stale connections. In handleDisconnect, verify the socket set is truly empty before emitting 'user-offline'. Add a periodic cleanup job.
